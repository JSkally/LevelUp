The full updated workflow:
```yaml
name: EAS Build
on:
  push:
    branches: [main]
    paths:
      - 'apps/mobile/**'
      - 'packages/**'

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: pnpm/action-setup@v4
        with:
          version: 9

      - uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: 'pnpm'

      - uses: expo/expo-github-action@v8
        with:
          eas-version: latest
          token: ${{ secrets.EXPO_TOKEN }}

      - run: pnpm install --frozen-lockfile

      - name: Build app
        run: eas build --platform all --non-interactive
        working-directory: apps/mobile

      - name: Publish OTA update
        run: eas update --channel production --message "Release from CI commit ${{ github.sha }}" --non-interactive
        working-directory: apps/mobile
```

Note: `eas update` requires the project to be linked to an EAS project (via `eas.json` update URL and project ID). The step will work once the developer runs `eas init` and sets the real project ID in eas.json and app.json. The CI step itself is structurally correct and will succeed once credentials are configured.
