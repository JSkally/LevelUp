    APPS/API/SRC/PLUGINS/TIEGATE.TS — per RESEARCH.md Pattern 8 exactly:
    ```typescript
    import { getAuth } from '@clerk/fastify'
    import { eq } from 'drizzle-orm'
    import { users, subscriptionTiers } from '@repo/db'
    import type { TierCapabilities } from '@repo/shared'

    export const requireTierCapability = (capability: keyof TierCapabilities) => {
      return async (request: FastifyRequest, reply: FastifyReply) => {
        const { userId } = getAuth(request)
        if (!userId) return reply.status(401).send({ error: 'Unauthorized' })

        const user = await request.server.db.query.users.findFirst({
          where: eq(users.clerkId, userId),
          with: { tier: true },
        })

        const caps = user?.tier?.capabilities as TierCapabilities | null
        if (!caps?.[capability]) {
          return reply.status(403).send({
            error: 'TIER_GATE',
            requiredCapability: capability,
            currentTier: user?.tier?.name ?? 'none',
            upgradeUrl: '/upgrade',
          })
        }
      }
    }
    ```

    useTierCapability hook (client-side):
    ```typescript
    'use client'
    import { useUser } from '@clerk/nextjs'
    import type { TierCapabilities } from '@repo/shared'

    export const useTierCapability = (capability: keyof TierCapabilities): boolean => {
      const { user } = useUser()
      // Tier capabilities stored in user.publicMetadata after server sets them
      const caps = user?.publicMetadata?.tierCapabilities as TierCapabilities | undefined
      return caps?.[capability] ?? false
    }
    ```

    Note on tier capability propagation: user.publicMetadata.tierCapabilities is set by the API when a user's tier is assigned. The webhook handler (Plan 02) sets role; a separate mechanism (called from PUT /api/users/:id/tier) sets tierCapabilities in publicMetadata. For Phase 1 demo purposes, seed a test user's tier in the DB and update their publicMetadata via the Clerk Dashboard.

    APPS/WEB/COMPONENTS/TIEGATE.TSX:
    ```typescript
    'use client'
    import { useTierCapability } from '@/lib/tiers'
    import type { TierCapabilities } from '@repo/shared'
    import Link from 'next/link'

    interface TierGateProps {
      capabilityKey: keyof TierCapabilities
      children: React.ReactNode
      upgradeMessage?: string
    }

    export function TierGate({ capabilityKey, children, upgradeMessage }: TierGateProps) {
      const hasCapability = useTierCapability(capabilityKey)

      if (!hasCapability) {
        return (
          <div className="relative">
            <div className="pointer-events-none opacity-40 select-none">{children}</div>
            <div className="absolute inset-0 flex flex-col items-center justify-center bg-background/80 rounded-lg">
              <span className="text-2xl mb-2">🔒</span>
              <p className="text-sm text-muted-foreground mb-3">
                {upgradeMessage ?? 'This feature requires a higher tier'}
              </p>
              <Link href="/upgrade" className="text-sm font-medium underline">
                Upgrade your plan
              </Link>
            </div>
          </div>
        )
      }

      return <>{children}</>
    }
    ```
